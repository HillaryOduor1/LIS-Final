import * as React from "react";
import { useSettings } from "../../../stores/settings-store";
import { BellIcon, Volume2, MailIcon } from "../../icons";

export const NotificationsTab: React.FC = () => {
  const { settings, updateSetting } = useSettings();

  const notificationTypes = [
    {
      id: "desktopNotifications",
      title: "Desktop Notifications",
      description: "Show notifications in your browser",
      icon: <BellIcon size={20} />,
      value: settings.notifications.desktopNotifications,
      action: () => updateSetting("notifications", {
        ...settings.notifications,
        desktopNotifications: !settings.notifications.desktopNotifications
      }),
    },
    {
      id: "sound",
      title: "Sound Alerts",
      description: "Play sounds for notifications",
      icon: <Volume2 size={20} />,
      value: settings.notifications.sound,
      action: () => updateSetting("notifications", {
        ...settings.notifications,
        sound: !settings.notifications.sound
      }),
    },
    {
      id: "emailNotifications",
      title: "Email Notifications",
      description: "Receive notifications via email",
      icon: <MailIcon size={20} />,
      value: settings.notifications.emailNotifications,
      action: () => updateSetting("notifications", {
        ...settings.notifications,
        emailNotifications: !settings.notifications.emailNotifications
      }),
    },
    {
      id: "pushNotifications",
      title: "Push Notifications",
      description: "Receive push notifications",
      icon: <BellIcon size={20} />,
      value: settings.notifications.pushNotifications,
      action: () => updateSetting("notifications", {
        ...settings.notifications,
        pushNotifications: !settings.notifications.pushNotifications
      }),
    },
  ];

  const notificationCategories = [
    { id: "security", label: "Security Alerts", description: "Important security updates" },
    { id: "updates", label: "System Updates", description: "App updates and maintenance" },
    { id: "messages", label: "Messages", description: "New messages and comments" },
    { id: "marketing", label: "Marketing", description: "Promotional emails and offers" },
  ];

  const frequencyOptions = [
    { id: "immediate", label: "Immediately" },
    { id: "hourly", label: "Hourly Digest" },
    { id: "daily", label: "Daily Digest" },
    { id: "weekly", label: "Weekly Digest" },
  ];

  return (
    <div className="space-y-8">
      {/* Global Toggle */}
      <div className="space-y-4">
        <div className="flex items-center justify-between p-4 border border-[var(--border)] rounded-lg bg-[var(--surface)]">
          <div>
            <div className="font-bold text-[var(--text)]">Enable Notifications</div>
            <div className="text-sm text-[var(--muted)]">
              Master switch for all notifications
            </div>
          </div>
          <button
            onClick={() => updateSetting("notifications", {
              ...settings.notifications,
              enabled: !settings.notifications.enabled
            })}
            className={`
              relative inline-flex h-6 w-11 items-center rounded-full transition-colors
              ${settings.notifications.enabled ? "bg-[var(--accent)]" : "bg-[var(--border)]"}
            `}
          >
            <span
              className={`
                inline-block h-4 w-4 transform rounded-full bg-white transition-transform
                ${settings.notifications.enabled ? "translate-x-6" : "translate-x-1"}
              `}
            />
          </button>
        </div>
      </div>

      {/* Notification Types */}
      <div className="space-y-4">
        <div>
          <h3 className="text-xl font-bold text-[var(--text)]">Notification Channels</h3>
          <p className="text-sm text-[var(--muted)]">Choose how you want to receive notifications</p>
        </div>

        <div className="space-y-3">
          {notificationTypes.map((type) => (
            <div
              key={type.id}
              className="flex items-center justify-between p-4 border border-[var(--border)] rounded-lg hover:bg-[var(--accent)]/5 transition-colors"
            >
              <div className="flex items-center gap-4">
                <div className="text-[var(--accent)]">{type.icon}</div>
                <div>
                  <div className="font-bold text-[var(--text)]">{type.title}</div>
                  <div className="text-sm text-[var(--muted)]">{type.description}</div>
                </div>
              </div>
              <button
                onClick={type.action}
                disabled={!settings.notifications.enabled}
                className={`
                  relative inline-flex h-6 w-11 items-center rounded-full transition-colors
                  ${type.value ? "bg-[var(--accent)]" : "bg-[var(--border)]"}
                  ${!settings.notifications.enabled ? "opacity-50 cursor-not-allowed" : ""}
                `}
              >
                <span
                  className={`
                    inline-block h-4 w-4 transform rounded-full bg-white transition-transform
                    ${type.value ? "translate-x-6" : "translate-x-1"}
                  `}
                />
              </button>
            </div>
          ))}
        </div>
      </div>

      {/* Notification Categories */}
      <div className="space-y-4">
        <div>
          <h3 className="text-xl font-bold text-[var(--text)]">Notification Categories</h3>
          <p className="text-sm text-[var(--muted)]">Select which types of notifications you want to receive</p>
        </div>

        <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
          {notificationCategories.map((category) => (
            <div
              key={category.id}
              className="flex items-center justify-between p-4 border border-[var(--border)] rounded-lg bg-[var(--surface)]"
            >
              <div>
                <div className="font-bold text-[var(--text)]">{category.label}</div>
                <div className="text-sm text-[var(--muted)]">{category.description}</div>
              </div>
              <input
                type="checkbox"
                checked={settings.notifications.categories?.includes(category.id)}
                onChange={() => {
                  const currentCategories = settings.notifications.categories || [];
                  const newCategories = currentCategories.includes(category.id)
                    ? currentCategories.filter((id: string) => id !== category.id)
                    : [...currentCategories, category.id];
                  updateSetting("notifications", {
                    ...settings.notifications,
                    categories: newCategories
                  });
                }}
                className="h-4 w-4 rounded border-[var(--border)] text-[var(--accent)] focus:ring-[var(--accent)] bg-[var(--bg)]"
              />
            </div>
          ))}
        </div>
      </div>

      {/* Frequency */}
      <div className="space-y-4">
        <div>
          <h3 className="text-xl font-bold text-[var(--text)]">Notification Frequency</h3>
          <p className="text-sm text-[var(--muted)]">How often you want to receive notifications</p>
        </div>

        <div className="grid grid-cols-2 md:grid-cols-4 gap-3">
          {frequencyOptions.map((frequency) => (
            <button
              key={frequency.id}
              onClick={() => updateSetting("notifications", {
                ...settings.notifications,
                frequency: frequency.id as any
              })}
              className={`
                  flex flex-col items-center justify-center p-4 border rounded-lg transition-all
                  ${settings.notifications.frequency === frequency.id
                  ? "border-[var(--accent)] bg-[var(--accent)]/5 text-[var(--accent)]"
                  : "border-[var(--border)] hover:border-[var(--accent)]/50 hover:bg-[var(--accent)]/5 text-[var(--muted)]"
                }
                `}
            >
              <span className="font-bold text-sm text-[var(--text)]">{frequency.label}</span>
            </button>
          ))}
        </div>
      </div>

      {/* Quiet Hours */}
      <div className="space-y-4">
        <div>
          <h3 className="text-xl font-bold text-[var(--text)]">Quiet Hours</h3>
          <p className="text-sm text-[var(--muted)]">Set times when you don't want to be disturbed</p>
        </div>

        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
          <div className="space-y-3">
            <label className="text-xs font-bold text-[var(--muted)] uppercase tracking-widest opacity-70">Start Time</label>
            <input
              type="time"
              defaultValue="22:00"
              className="w-full px-3 py-2 border border-[var(--border)] rounded-md bg-[var(--bg)]"
            />
          </div>

          <div className="space-y-3">
            <label className="text-xs font-bold text-[var(--muted)] uppercase tracking-widest opacity-70">End Time</label>
            <input
              type="time"
              defaultValue="07:00"
              className="w-full px-3 py-2 border border-[var(--border)] rounded-md bg-[var(--bg)]"
            />
          </div>
        </div>
      </div>

      {/* Test Notification */}
      <div className="p-4 border border-[var(--border)] rounded-lg bg-[var(--accent)]/5">
        <div className="flex items-center justify-between">
          <div>
            <div className="font-bold text-[var(--text)]">Test Notification</div>
            <div className="text-sm text-[var(--muted)]">
              Send a test notification to verify your settings
            </div>
          </div>
          <button
            onClick={() => {
              if (Notification.permission === "granted") {
                new Notification("Test Notification", {
                  body: "This is a test notification from the settings page.",
                  icon: "/assets/logo.svg",
                });
              } else if (Notification.permission !== "denied") {
                Notification.requestPermission().then(permission => {
                  if (permission === "granted") {
                    new Notification("Test Notification", {
                      body: "This is a test notification from the settings page.",
                      icon: "/assets/logo.svg",
                    });
                  }
                });
              }
            }}
            className="px-4 py-2 bg-[var(--accent)] text-white rounded-md hover:filter hover:brightness-110 transition-colors"
          >
            Send Test
          </button>
        </div>
      </div>
    </div>
  );
};
